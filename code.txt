#include <iostream>

using namespace std;
int main() {

    int n;

    cout << "Enter a positive integer: ";
    cin >> n;
    int i = 1;

    while (i <= n) {

        int j = 1;

        while (j <= n) {

            // Perform some operation

            cout << "(" << i << ", " << j << ") ";

            j *= 2; // Logarithmic growth

        }

        cout << endl;

        i++;

    }



    return 0;

}
